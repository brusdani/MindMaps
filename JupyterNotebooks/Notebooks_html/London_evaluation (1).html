from sklearn.metrics.pairwise import cosine_similarity
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt

cos_sim_matrix = cosine_similarity(embeddings_london)


plt.figure(figsize=(8, 6))
sns.heatmap(cos_sim_matrix, cmap='YlGnBu', annot=True, fmt='.2f', xticklabels=[f'London{i+1}.txt' for i in range(10)], yticklabels=[f'London{i+1}.txt' for i in range(10)])
plt.title("Cosine Similarity Matrix: London")
plt.show()
