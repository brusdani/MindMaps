from sklearn.metrics.pairwise import cosine_similarity
import numpy as np

# Function to calculate pairwise cosine similarity for each cluster
def calculate_similarity_within_clusters(embeddings, clusters):
    similarity_scores = {}

    # Get the unique clusters (0 for Churchill, 1 for Havel)
    unique_clusters = np.unique(clusters)

    for cluster in unique_clusters:
        # Get embeddings of the current cluster
        cluster_embeddings = embeddings[clusters == cluster]

        # Calculate pairwise cosine similarity within this cluster
        cosine_sim = cosine_similarity(cluster_embeddings)

        # Store the similarity scores for this cluster
        similarity_scores[cluster] = cosine_sim

    return similarity_scores

# Calculate similarity within each cluster
similarity_scores = calculate_similarity_within_clusters(embeddings, clusters)

# Print similarity for each cluster
for cluster, scores in similarity_scores.items():
    print(f"Cluster {cluster} pairwise similarity matrix:\n")
    print(scores)  # This will print the cosine similarity matrix for the cluster
    print("\n")

def calculate_average_similarity(similarity_scores):
    average_similarities = {}

    for cluster, sim_matrix in similarity_scores.items():
        # The diagonal of the similarity matrix represents similarity with itself (which is always 1)
        # So we ignore the diagonal elements to calculate the average similarity
        triu_sim = np.triu(sim_matrix, k=1)  # Only upper triangle to avoid double counting
        non_diag_sim = triu_sim[triu_sim != 0]  # Get the non-zero values
        average_similarity = np.mean(non_diag_sim)

        average_similarities[cluster] = average_similarity

    return average_similarities

# Calculate average similarity for each cluster
average_similarities = calculate_average_similarity(similarity_scores)

# Print the average similarity for each cluster
for cluster, avg_sim in average_similarities.items():
    print(f"Average similarity for Cluster {cluster}: {avg_sim}")
